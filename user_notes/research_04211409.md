## 2.nginxコンフィグ
機微情報は`**secret**`でマスク処理済。
```bash
[opc@ls-gmm-instance ~]$ sudo cat /etc/nginx/conf.d/lsapp.conf
# /etc/nginx/conf.d/lsapp.conf

# 80: HTTPSへ(Cloudflareプロキシでも一応残す)
server {
    listen 80;
    listen [::]:80;
    server_name **secret**;

    location / {
        return 301 https://$host$request_uri;
    }
}

# 443: Cloudflare Origin CA + 必要パスのみFlaskへ
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name **secret**;

    ssl_certificate     /etc/nginx/cf-origin/origin.crt;
    ssl_certificate_key /etc/nginx/cf-origin/origin.key;

    # 既存のLet's Encrypt向け設定は"必須ではない"が、ファイルが存在するなら残しても害は少ない
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # OAuth認証リンク処理
    location ^~ /oauth/ {
        proxy_pass http://127.0.0.1:8080;
        proxy_http_version 1.1;

        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_intercept_errors on;
        error_page 502 503 504 =200 /_oauth_down;
    }

    location = /_oauth_down {
        add_header Content-Type text/html;
        return 200 "<!doctype html><meta charset=utf-8><title>OAuth</title><h1>OAuth</h1><p>認証処理の待受サーバが起動していません。GmailMonitor(app.py)実行中に再度リンクを開いてください。</p>";
    }

    # 1) デフォルトは404で隠蔽(Flaskが落ちていても502にしない)
    location / {
        add_header Content-Type text/plain;
        return 404 "Not Found\n";
    }

    # 2) /user (UI) はアプリへ
    location ^~ /user {
        proxy_pass http://127.0.0.1:8080;
        proxy_http_version 1.1;

        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Flaskが落ちているときはCloudflare 502ではなく自前503へ
        proxy_intercept_errors on;
        error_page 502 503 504 = /_user_down;
    }

    location = /_user_down {
        add_header Content-Type text/html;
        return 503 "<!doctype html><meta charset=utf-8><title>Unavailable</title><h1>Unavailable</h1><p>GmailMonitor is not running now.</p>";
    }

    # 3) /api/me (API) もアプリへ
    location ^~ /api/me {
        proxy_pass http://127.0.0.1:8080;
        proxy_http_version 1.1;

        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_intercept_errors on;
        error_page 502 503 504 = /_api_down;
    }

    location = /_api_down {
        add_header Content-Type application/json;
        return 503 '{"error":"service temporarily unavailable"}';
    }

    # 4) /admin は予約(今は隠す)。将来ここをアプリに通すなら、このlocationをproxyに差し替える
    location ^~ /admin {
        return 404;
    }
}
```
### 注記：
- `**secret**`は共にサーバーURLで、正しく設定されているように見える。
